title GenerateDailyPlan

participant Agronomist
participant DREAM
participant Farmer

activate Agronomist
Agronomist->DREAM:request_login_page()
activate DREAM
Agronomist<<--DREAM:login page
opt page received
Agronomist->DREAM:login()
Agronomist<<--DREAM:answer
opt logged in
Agronomist<<--DREAM:generate_daily_plan()
alt daily plan is not ok
loop 
Agronomist->DREAM:change_daily_plan()
activate DREAM
DREAM->DREAM: check_schedule()
deactivate DREAM
Agronomist<<--DREAM: answer
end
else daily plan is ok
Agronomist->DREAM:confirm_plan()
end
deactivate Agronomist
DREAM->Farmer: notify_meeting()
deactivate DREAM
end
end
